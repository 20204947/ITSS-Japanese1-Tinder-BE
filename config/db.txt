CREATE TABLE `User` (
    `userID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `email` VARCHAR(255) NOT NULL UNIQUE,
    `password` VARCHAR(255) NOT NULL,
    `name` VARCHAR(255) NOT NULL,
    `gender` INT NOT NULL,
    `dob` DATE NOT NULL,
    `role` BIGINT NOT NULL,
    `imageURL` VARCHAR(255) NULL,
    `firstFavouriteID` BIGINT UNSIGNED NULL,
    `secondFavouriteID` BIGINT UNSIGNED NULL,
    `thirdFavouriteID` BIGINT UNSIGNED NULL,
    `fourthFavouriteID` BIGINT UNSIGNED NULL,
    `fifthFavouriteID` BIGINT UNSIGNED NULL
);

CREATE TABLE `Favourite` (
    `favouriteID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `favouriteName` VARCHAR(255) NOT NULL
);

CREATE TABLE `Action` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `userID` BIGINT UNSIGNED NOT NULL,
    `targetUserID` BIGINT UNSIGNED NOT NULL,
    `action` INT NOT NULL,
);

CREATE TABLE `Matching` (
    `matchingID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `userA` BIGINT UNSIGNED NOT NULL,
    `userB` BIGINT UNSIGNED NOT NULL,
    `status` BIGINT NOT NULL
);

CREATE TABLE `Event` (
    `eventID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `eventName` VARCHAR(255) NOT NULL,
    `eventTime` TIME NOT NULL
);

CREATE TABLE `JoinEvent` (
    `joinEventID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `userID` BIGINT NOT NULL,
    `eventID` BIGINT NOT NULL,
);

CREATE TABLE `Message` (
    `messageID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `matchingID` BIGINT UNSIGNED NULL,
    `fromUserID` BIGINT UNSIGNED NOT NULL,
    `time` TIME NOT NULL,
    `context` BIGINT NOT NULL
);
